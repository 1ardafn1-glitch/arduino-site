Temel Kod — Sadece Potansiyometre ile Kontrol 
/*
  ============================================================
   IRFZ44N MOSFET — LED / Fan Hız Kontrol Devresi
   Kart    : Arduino Uno / Nano / Mega
   Kontrol : Potansiyometre → PWM → IRFZ44N Gate
  ============================================================
*/

// ─── Pin Tanımları ──────────────────────────────────────────
const int POT_PIN   = A0;   // Potansiyometre orta ucu
const int MOSFET_PIN = 9;    // PWM pini → IRFZ44N Gate (3,5,6,9,10,11 kullan)

// ─── Yumuşatma (Gürültü azaltma) ────────────────────────────
const int ORNEK_SAYISI = 10;  // Titreşimi azaltmak için N okuma ortalaması
int okumalar[10];
int okumaIndex = 0;
long toplam = 0;

void setup() {
  Serial.begin(9600);
  pinMode(MOSFET_PIN, OUTPUT);
  analogWrite(MOSFET_PIN, 0);  // Başlangıçta çıkış KAPALI

  // Yumuşatma dizisini sıfırla
  for (int i = 0; i < ORNEK_SAYISI; i++) okumalar[i] = 0;

  Serial.println("=== IRFZ44N Kontrolcu Hazir ===");
  Serial.println("POT_HAM | PWM_CIKIS | GOREV_%");
}

void loop() {
  // ── Potansiyometre oku ve yumuşat ────────────────────────
  toplam -= okumalar[okumaIndex];
  okumalar[okumaIndex] = analogRead(POT_PIN);
  toplam += okumalar[okumaIndex];
  okumaIndex = (okumaIndex + 1) % ORNEK_SAYISI;

  int potDegeri = toplam / ORNEK_SAYISI;

  // ── PWM'e dönüştür (0–255) ───────────────────────────────
  int pwmDegeri = map(potDegeri, 0, 1023, 0, 255);
  analogWrite(MOSFET_PIN, pwmDegeri);

  // ── Seri Monitör çıktısı ─────────────────────────────────
  float gorevYuzdesi = (pwmDegeri / 255.0) * 100.0;
  Serial.print("POT: ");      Serial.print(potDegeri);
  Serial.print("  |  PWM: "); Serial.print(pwmDegeri);
  Serial.print("  |  GOREV: "); Serial.print(gorevYuzdesi, 1);
  Serial.println("%");

  delay(100);
}
Bonus Versiyon — NTC Termistör Sıcaklık Korumasıyla
💡 Bonus Özellik: A1 pinine bağlı 10kΩ NTC termistör sıcaklığı izler. Belirlediğin eşiği (örn. 40°C) aşarsa fan, pot'tan bağımsız olarak %100 hıza zorlanır. PC fanı veya elektronik soğutma kutusu için mükemmel! 
/*
  ============================================================
   IRFZ44N — NTC Sıcaklık Korumalı Fan Kontrolcüsü (Bonus!)
   Pinler:
     A0 → Potansiyometre
     A1 → NTC Termistör (10kΩ bölücü, GND'ye)
     D9  → IRFZ44N Gate (PWM)
  ============================================================
*/

const int POT_PIN     = A0;
const int SICAKLIK_PIN = A1;
const int MOSFET_PIN  = 9;

const float SICAKLIK_ESIGI = 40.0;  // °C — fani %100'e zorlayan esik
const float BETA           = 3950.0; // NTC Beta katsayisi
const float R_SABIT        = 10000.0;// Seri direnc: 10kΩ
const float R_NOMINAL      = 10000.0;// NTC direnci @ 25°C
const float T_NOMINAL      = 25.0;   // NTC nominal sicaklik (°C)

// Yumuşatma
const int ORNEK = 10;
int potBuf[10]; long potToplam = 0; int potIdx = 0;

void setup() {
  Serial.begin(9600);
  pinMode(MOSFET_PIN, OUTPUT);
  analogWrite(MOSFET_PIN, 0);
  for (int i = 0; i < ORNEK; i++) potBuf[i] = 0;
  Serial.println("=== IRFZ44N Fan + Sicaklik Kontrolcusu ===");
}

float sicaklikOku() {
  int ham = analogRead(SICAKLIK_PIN);
  float direnc = R_SABIT * ((1023.0 / ham) - 1.0);
  float steinhart = direnc / R_NOMINAL;
  steinhart = log(steinhart);
  steinhart /= BETA;
  steinhart += 1.0 / (T_NOMINAL + 273.15);
  steinhart = 1.0 / steinhart;
  return steinhart - 273.15;  // Kelvin → Celsius
}

void loop() {
  // ── Potansiyometre yumuşatması ───────────────────────────
  potToplam -= potBuf[potIdx];
  potBuf[potIdx] = analogRead(POT_PIN);
  potToplam += potBuf[potIdx];
  potIdx = (potIdx + 1) % ORNEK;
  int potDegeri = potToplam / ORNEK;

  // ── Sıcaklık oku ─────────────────────────────────────────
  float sicaklik = sicaklikOku();

  // ── PWM çıkışını belirle ─────────────────────────────────
  int pwmDegeri;
  bool sicaklikKorumasi = false;

  if (sicaklik >= SICAKLIK_ESIGI) {
    pwmDegeri = 255;   // Maksimum hıza zorla!
    sicaklikKorumasi = true;
  } else {
    pwmDegeri = map(potDegeri, 0, 1023, 0, 255);
  }

  analogWrite(MOSFET_PIN, pwmDegeri);

  // ── Seri Monitör çıktısı ─────────────────────────────────
  Serial.print("POT: ");        Serial.print(potDegeri);
  Serial.print("  SICAKLIK: "); Serial.print(sicaklik, 1); Serial.print("C");
  Serial.print("  PWM: ");      Serial.print(pwmDegeri);
  Serial.print("  GOREV: ");    Serial.print((pwmDegeri/255.0)*100.0, 1); Serial.print("%");
  if (sicaklikKorumasi) Serial.print("  !! SICAKLIK KORUMASI AKTIF!");
  Serial.println();

  delay(100);
}
⚠️ Önemli: IRFZ44N bir N-kanal MOSFET'tir. 5V gate geriliminde çalışır fakat en iyi performans ≥10V'da elde edilir. 5V'da 5A'ya kadar yüklerde kullanılabilir. Gate ile Source arasına 10kΩ pull-down, Gate serisine 100Ω gate direnci eklemeyi unutma!

🔬 PWM Kontrolü Nasıl Çalışır?

PWM (Darbe Genişlik Modülasyonu), MOSFET gate'ini ~490Hz'de hızla AÇIK ve KAPALI konumuna getirir. AÇIK kalma oranını (görev döngüsü) değiştirerek yüke ulaşan ortalama gücü kontrol ederiz.

%0
MOSFET hep KAPALI
Yük = 0W
%50
Yarı süre açık
Yük ≈ %50 güç
%100
Hep AÇIK
Yük = tam güç

Arduino'nun analogWrite(pin, 0–255) fonksiyonu bunu kontrol eder. map(potDegeri, 0, 1023, 0, 255) ise 10-bitlik ADC okumasını 8-bitlik PWM değerine dönüştürür.

💡 Ekleyebileceğin Bonus Özellikler
🌡️
NTC Termistör Sıcaklık Koruması
Sıcaklık eşiği aşıldığında fanı otomatik olarak %100'e zorla. Soğutma kutuları için harika. (Bonus kodda mevcut!)
📺
OLED Ekran
SSD1306 I2C OLED ekran ekleyerek pot değerini, PWM yüzdesini ve sıcaklığı görsel olarak göster. Seri Monitör'e gerek kalmaz.
🔘
Mod Butonu
Buton ekleyerek Manuel (pot) modu ile Otomatik (sıcaklık) modu arasında geçiş yap. LED göstergesi ile aktif modu göster.
📱
Bluetooth Kontrolü
HC-05 Bluetooth modülü ekleyerek fan hızını telefonundan kontrol et. Serial Bluetooth Terminal uygulaması ile kullanılabilir.
⚠️ Güvenlik Uyarıları:
  • IRFZ44N'nin 55V ve 49A sınırını asla aşma.
  • Sürekli 1–2A üzeri akımlarda soğutucu (heatsink) kullan.
  • Motor/fan gibi yüklerde mutlaka flyback diyot (1N4007) ekle.
  • Fanları Arduino'nun 5V pininden besleme; harici güç kaynağı kullan!
  • 100Ω gate direnci, PWM titreşimini önler ve Arduino pinini korur.
